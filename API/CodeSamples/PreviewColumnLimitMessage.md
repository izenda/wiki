#PreviewColumnLimitMessage

[[_TOC_]]

##About

Gets or sets the message displayed when the [[PreviewColumnLimit|/API/CodeSamples/PreviewColumnLimit]] is exceeded.

**Default Value:** 

Reports with more than 100 fields must be exported to CSV. Please check CSV Export to view data.

//main class: inherits DatabaseAdHocConfig or FileSystemAdHocConfig
public class CustomAdHocConfig : Izenda.AdHoc.DatabaseAdHocConfig
{
  // Configure settings
  // Add custom settings after setting the license key and connection string by overriding the ConfigureSettings() method
  public static void InitializeReporting() {
    //Check to see if we've already initialized.
    if (HttpContext.Current.Session == null || HttpContext.Current.Session["ReportingInitialized"] != null)
      return;
    AdHocSettings.LicenseKey = "INSERT_LICENSE_KEY_HERE";
    AdHocSettings.SqlServerConnectionString = "INSERT_CONNECTION_STRING_HERE";
    Izenda.AdHoc.AdHocSettings.AdHocConfig = new CustomAdHocConfig();
    AdHocSettings.PreviewColumnLimitMessage = "Use less columns!!!"; //The relevant setting
    HttpContext.Current.Session["ReportingInitialized"] = true;
  }
}
