#DashboardsCss

[[_TOC_]]

##About

_**Obsolete:** Use DashboardsCssUrl instead._

Gets or sets the CSS style for the Izenda.Web.UI.TabStrip.

**Default Value:** @import url(rp.aspx?auid=[auid]&css=dashboard);

public class CustomAdHocConfig : Izenda.AdHoc.DatabaseAdHocConfig
{

  public static void InitializeReporting() {
    //Check to see if we've already initialized.
    if (AdHocContext.Initialized)
      return;
    AdHocSettings.LicenseKey = "INSERT_LICENSE_KEY_HERE";
    AdHocSettings.SqlServerConnectionString = "INSERT_CONNECTION_STRING_HERE";
    Izenda.AdHoc.AdHocSettings.AdHocConfig = new CustomAdHocConfig();
    AdHocSettings.DashboardsCss = "@import url(reporting/rp.aspx?auid=[auid]&css=dashboard);"; //The relevant setting
    AdHocContext.Initialized = true;
  }
}
